\subsection{复杂案例}
\begin{enumerate}
    \item 基本不等式：$\forall a_1,a_2,\cdots，a_n$，有%\forall任意，_下标，_{}下标不止一个，如aij，\dot一个点，\dots三个点居下，\cdots三个点居中，\vdots竖向省略号
    \begin{equation*}
        \frac{a_1+a_2+\cdots+a_n}{n}\geq\sqrt[n]{a_1a_2\cdots a_n}%geq大于等于，req小于等于，neq不等于
    \end{equation*}
    当且仅当$a_1=a_2=\cdots=a_n$等号成立。
    \item vondermonde行列式：
    \begin{equation*}
        |\bm{A}|=\begin{vmatrix}%\bm加粗
        1 & 1&\cdots&1\\
        a_1&a_2&\cdots&a_n\\
        a_1^2 &a_2^2 &\cdots &a_n^2\\
        \vdots &\vdots & &\vdots\\
        a_1^{n-1} & a_2^{n-1} &\cdots &a_n^{n-1}
        \end{vmatrix}
        =\prod_{1\leq j<i\leq n }(a_i-a_j)%\prod累乘，\sum累加，\int积分，_后面跟下面的， ^后面跟上面的
    \end{equation*}
    \item 定积分的定义：若$f$是定义在$[a,b]$上的函数，$J$是一个确定的实数，若$\forall \varepsilon >0$，$\exists \delta>0$，使得对$[a,b]$以及任意介电集$\{\xi_i\}$，只要$\|T\|<\delta$，就有
    \begin{equation*}
        \left|\sum_{i=1}^nf(\xi_i)\Delta x_i-J\right|<\varepsilon\,.
        %在||，（），[],{}前加\left或\right，那么就可以让符号大小随内容变化
    \end{equation*}
    则称$f$在$[a,b]$上Riemann可积，称$J$是$f$在$[a,b]$上的定积分，记作
    \begin{equation*}
        J=\int_a^bf(x)\mathrm{d}x\,.
    \end{equation*}
    即
    \begin{equation*}
        J=\lim_{\|T\| \to 0}\sum_{i=1}^nf(\xi_i)\Delta x_i=\int _a^bf(x)\mathrm{d}x\,.%\|双竖杠
    \end{equation*}
    \item 线性变换的定义：数域$P$上的线性空间$V$上的一个线性变换$\mathscr{A}$ 称为线性变换，如果$\forall\bm\alpha ,\bm\beta \in V,k\in P$，都有
    \begin{equation*}
        \mathscr{A}(\bm\alpha+\bm\beta)=\mathscr{A}(\bm\alpha)+\mathscr{A}(\bm\beta),
        \mathscr{A}(k\bm\alpha)=k\mathscr{A}(\bm\alpha)\,.
    \end{equation*}
    %\mathscr数学花体
    \item 设 $\mathbb{R}^3$中的点$A(1,0,-1)$还有$B(0,-2,1)$，则$\overrightarrow{AB}=%overrightarrow上面右箭头，overleftarrow上面左箭头
    \begin{pmatrix}
    -1\\
    -2\\
    2
    \end{pmatrix}$\,.
    \item 设函数
    \begin{equation*}
        f(x)=\begin{cases}
        x\sin \frac{1}{x} &x\neq 0\,,\\
        0 &x=0 \,.%&的本质是将一行划分为多个“列”，每一列由下一个&或行尾分隔。
        \end{cases}
    \end{equation*}
    由于 $\displaystyle\lim_{x\to 0}x\sin \frac{1}{x}=0=f(0)$，故$f$在$x=0$处连续，但连续不一定可导，下面证明$f$在$x=0$处\\[5pt]不可导，
    
    由于
    \begin{equation*}
        \lim_{\Delta x\to 0}\frac{f(0+\Delta x)-f(0)}{\Delta x}=lim_{\Delta x\to 0}\sin\frac{1}{\Delta x}\,.
    \end{equation*}
    此极限不存在，故$f^\prime(0)$ 不存在，即$f$在$x=0$处不可导。
    %\prime对应'
\end{enumerate}
